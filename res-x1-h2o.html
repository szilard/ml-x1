d <- read_delim("res-x1-h2o.csv", delim=":", 
        col_names = c("instance","tool","cpucores","runtime")) 
## Parsed with column specification:
## cols(
##   instance = col_character(),
##   tool = col_character(),
##   cpucores = col_character(),
##   runtime = col_double()
## )
cpucores_order <- unique(d$cpucores)
d %>% mutate(cpucores=factor(cpucores, levels=cpucores_order)) %>% 
  filter(!cpucores %in% c("0","0-1","0-3")) %>% 
  ggplot() + geom_boxplot(aes(x = cpucores, y = runtime)) + ylim(c(0,NA)) +
  coord_flip()

d %>% mutate(cpucores=factor(cpucores, levels=cpucores_order)) %>% 
  filter(cpucores %in% c("0","0-1","0-3","0-7","0-15","0-31","0-63")) %>% 
  mutate(ncores = as.numeric(gsub(".*-","",cpucores))+1) %>% 
  mutate(runtime_norm = runtime*ncores/min(runtime*ncores)) %>% 
  ggplot() + geom_boxplot(aes(x = cpucores, y = runtime_norm)) + ylim(c(0,NA))
